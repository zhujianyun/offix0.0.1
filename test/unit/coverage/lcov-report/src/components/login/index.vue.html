<template>
  <div class="body">
    <div class="bigbox">
      <div class="head">账户密码登录</div>
      <div><input type="text" v-model="userName" ref="userName" placeholder="手机/邮箱/账号" class="name"></div>
      <div><input type="password" v-model="passWord" ref="passWord" placeholder="输入密码" class="password"
                  @keyup.enter="login"></div>
      <el-checkbox v-model="checked" style="color:#a0a0a0;" class="next">下次自动登录</el-checkbox>
      <button type="button" @click="login" ref="enter" class="enter">登 录</button>
    </div>
  </div>
</template>
<script>
  import {setCookie, getCookie} from "../../api/cookie"
 
  export default {
    data() {
      return {
        userName: '',
        passWord: '',
        checked: false
      }
    },
    mounted() {
      if (getCookie('Uname')) {
        this.userName = getCookie('Uname');
      }
      if (getCookie('Pkid')) {
        this.passWord = getCookie('Pkid');
      }
      if (this.userName && this.passWord) {
        this.$refs.passWord.focus();
      }
    },
    methods: {
      login() {
        if (this.userName == "" || this.passWord == "") {
          this.$message.warning("请输入用户名或密码");
          return;
        } else {
          var name = this.userName;
          var pass = this.passWord;

          let data = {"Uname": this.userName, "Pwd": this.passWord};
          this.$HTTP('post', '/System_userLogin', data).then(res => {
            let pkid = res.result.Pkid;
            if (res.code == "200") {
              setCookie("Pkid", pkid, 1000 * 60);
              setCookie("Pwd",this.passWord, 1000 * 60)
              this.$message.warning('登录成功');
              this.$router.push('/staff');
            } else {
              this.$message.warning("账号或密码错误")
            }
          })
        }
        if (this.checked) {
          this.setCookie(name, pass, 7)
        }
      },
      setCookie(c_name, c_pwd, exdays) {
        var exdate = new Date();//获取时间
        exdate.setTime(exdate.getTime() + 24 * 60 * 60 * 1000 * exdays);//保存的天数
        //字符串拼接cookie
        window.document.cookie = "userName" + "=" + c_name + ";path=/;expires=" + exdate.toGMTString();
        window.document.cookie = "userPwd" + "=" + c_pwd + ";path=/;expires=" + exdate.toGMTString();
      },
      //读取cookie
      getCookie: function () {
        if (document.cookie.length > 0) {
          var arr = document.cookie.split('; ');//这里显示的格式需要切割一下自己可输出看下
          for (var i = 0; i < arr.length; i++) {
            var arr2 = arr[i].split('=');//再次切割
            //判断查找相对应的值
            if (arr2[0] == 'userName') {
              this.ruleForm.userName = arr2[1];//保存到保存数据的地方
            } else if (arr2[0] == 'userPwd') {
              this.ruleForm.password = arr2[1];
            }
          }
        }
      },
      //清除cookie
      clearCookie: function () {
        this.setCookie("", "", -1);//修改2值都为空，天数为负1天就好了
      }
    },
    created() {
      if (getCookie('userPwd')) {
        this.$router.push('/home');
      }
    }
  }
</script>
